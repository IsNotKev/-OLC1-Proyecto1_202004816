
INICIO -> LLAVELEFT INSTRUCCIONES LLAVERIGHT;

INSTRUCCIONES ->  INSTRUCCIONES INSTRUCCION
                | INSTRUCCION

INSTRUCCION ->    CONJUNTO 
                | EXPRESION PUNTOYCOMA
                | ENTRADA

CONJUNTO ->       CONJ DOSPUNTOS ID FLECHA TCONJ PUNTOYCOMA

TCONJ ->          CARACTER VIRGULILLA CARACTER
                | LISTA

LISTA ->          LISTA COMA CARACTER
                | CARACTER

EXPRESION ->      PUNTO EXPRESION EXPRESION 
                | OR EXPRESION EXPRESION
                | MAS EXPRESION EXPRESION
                | POR EXPRESION EXPRESION
                | INTERROGACION EXPRESION EXPRESION
                | ID
                | CADENA
                | CARACTER

ENTRADA ->        ID DOSPUNTOS CADENA PUNTOYCOMA
